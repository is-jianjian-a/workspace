定义需求信号词典
从内容中提取4类真实需求信号：痛点、决策、求助、行为
# 需求信号词典 pain_signals = ['崩溃', '焦虑', '后悔', '踩雷', '避雷', '智商税', '翻车', '失败', '痛苦', '煎熬', '折磨', '难', '累', '烦', '慌', '担心', '害怕', '无助', '孤独', '抑郁'] decision_signals = ['选哪个', '还是', '对比', '有必要吗', '值不值', '要不要', '买不买', '去不去', '怎么选', '纠结'] seeking_signals = ['求推荐', '求分享', '求攻略', '求经验', '怎么', '怎么办', '怎么做', '怎么解决', '如何', '怎样', '求助', '求救'] action_signals = ['记录', '打卡', '整理', '统计', '算', '查', '搜', '问', '对比', '备忘', '提醒', '设置', '下载', '收藏']
💡 信号设计逻辑
痛点信号：负面情绪词 → 暴露真实困境
决策信号：对比/选择词 → 消费/行为决策前
求助信号：疑问/请求词 → 主动寻求帮助
行为信号：动作词 → 当前使用的工具/方法
信号匹配与统计
遍历所有内容，匹配需求信号，统计各类信号的覆盖率和互动表现
# 信号提取算法 for signal_name, signals in [('痛点', pain_signals), ('决策', decision_signals), ('求助', seeking_signals), ('行为', action_signals)]: matched = [] for content in all_content: has_signal = any(s in content['text'] for s in signals) if has_signal: matched.append(content) # 统计词频 word_freq = {} for content in matched: for s in signals: if s in content['text']: word_freq[s] = word_freq.get(s, 0) + 1 print(f"{signal_name}: {len(matched)}条 ({len(matched)/len(all_content)*100:.1f}%)") print(f" 高频词: {sorted(word_freq.items(), key=lambda x: x[1], reverse=True)[:5]}")

🔍Step 4: 内容质量评估

定义内容类型判断规则
基于营销信号和UGC信号，将内容分为4类：纯营销、混合、纯UGC、中性
# 内容类型判断规则 marketing_signals = ['好物', '推荐', '测评', '必买', '清单', '品牌', '链接', '优惠', '团购', '种草', '带货', '广告', '赞助'] ugc_signals = ['求助', '怎么办', '正常吗', '崩溃', '焦虑', '后悔', '吐槽', '日记', '记录', '日常', 'vlog', '我的', '我家', '我娃'] def classify_content(text): has_marketing = any(s in text for s in marketing_signals) has_ugc = any(s in text for s in ugc_signals) if has_marketing and not has_ugc: return '纯营销' elif has_ugc and not has_marketing: return '纯UGC' elif has_marketing and has_ugc: return '混合' else: return '中性'
💡 判断逻辑
纯营销：只有商业推广信号，无个人叙事 → 可信度低
纯UGC：只有个人叙事信号，无商业推广 → 可信度极高
混合：两者兼有 → 可能是真实体验后的推荐，需进一步判断
中性：两者皆无 → 可能是知识分享或纯记录
内容分类与统计
对所有内容进行分类，统计各类型的分布和互动表现
# 内容分类统计 type_counts = {} type_interactions = {} for content in all_content: content_type = classify_content(content['text']) type_counts[content_type] = type_counts.get(content_type, 0) + 1 if content_type not in type_interactions: type_interactions[content_type] = [] type_interactions[content_type].append(content['total_interaction']) # 输出统计结果 for content_type, count in sorted(type_counts.items(), key=lambda x: x[1], reverse=True): avg_interaction = sum(type_interactions[content_type]) / len(type_interactions[content_type]) print(f"{content_type}: {count}条 ({count/len(all_content)*100:.1f}%), 平均互动{avg_interaction:.0f}")

💎Step 5: 高价值内容识别

# 高价值内容筛选标准 def identify_high_value(content): value_factors = [] # 互动阈值 if content['total_interaction'] > 5000: value_factors.append('高互动') # 评论阈值（共鸣指标） if content['comment_count'] > 1000: value_factors.append('高评论') # 分享阈值（传播指标） if content['share_count'] > 1000: value_factors.append('高分享') # 收藏阈值（实用指标） if content['collected_count'] > 1000: value_factors.append('高收藏') # 需求信号 if any(s in content['text'] for s in pain_signals): value_factors.append('含痛点') if any(s in content['text'] for s in seeking_signals): value_factors.append('含求助') return value_factors # 筛选高价值内容 high_value = [c for c in all_content if classify_content(c['text']) in ['纯UGC', '混合'] and c['total_interaction'] > 5000]
