情感推断逻辑

在从App Store获取评论时，我基于星级评分推断情感标签：

Python 
import requests
import json
import re
from datetime import datetime

# 定义AI相关关键词
AI_KEYWORDS = [
    'AI', '人工智能', '智能', '自动', '字幕', '翻译', '推荐', '算法',
    '识别', '语音', '画质', '增强', '总结', '摘要', '搜索', '滤镜',
    '特效', '美颜', '剪辑', '生成', '理解', '分析', '优化'
]

# 情感推断函数
def infer_sentiment(rating):
    """基于星级推断情感"""
    if rating <= 2:
        return "negative"  # 1-2星 = 负面
    elif rating == 3:
        return "neutral"   # 3星 = 中性
    elif rating >= 4:
        return "positive"  # 4-5星 = 正面
    return "neutral"

# 判断是否为痛点
def is_pain_point(content, rating):
    """判断是否为痛点（高优先级负面）"""
    pain_keywords = ['垃圾', '恶心', '难用', '卡顿', '闪退', 'bug', 
                     '广告太多', '无法', '不能', '失望', '卸载']
    if rating <= 2 and any(kw in content for kw in pain_keywords):
        return True
    return False

# 判断是否为期望
def is_expectation(content):
    """判断是否为功能期望"""
    expect_keywords = ['希望', '想要', '如果能', '建议增加', '期待', 
                       '需要', '最好', '为什么不', '怎么没有']
    return any(kw in content for kw in expect_keywords)

# 从App Store评论提取VOC
def extract_voc_from_review(review, app_name, app_id):
    """从单条评论提取VOC"""
    content = review.get('review', '')
    rating = review.get('rating', 3)

    # 检查是否AI相关
    is_ai_related = any(kw in content for kw in AI_KEYWORDS)

    # 推断情感
    sentiment = infer_sentiment(rating)

    # 进一步细分
    if is_pain_point(content, rating):
        sentiment = "pain_point"
    elif is_expectation(content):
        sentiment = "expectation"

    return {
        "id": f"voc_appstore_{len(all_vocs)+1:06d}",
        "platform": "app_store",
        "source_url": f"https://apps.apple.com/app/id{app_id}",
        "author": review.get('userName', 'anonymous'),
        "date": review.get('date', 'unknown'),
        "content": content,
        "context": f"{app_name} App Store评论",
        "sentiment": sentiment,
        "category": "other",  # 后续分析时再分类
        "ai_related": is_ai_related,
        "metadata": {
            "rating": rating,
            "app_name": app_name,
            "app_id": app_id
        }
    }

实际执行的App Store VOC收集代码

完整采集脚本
Python 
import requests
import json
import time
from datetime import datetime

# AI相关关键词
AI_KEYWORDS = [
    'AI', '人工智能', '智能', '自动', '字幕', '翻译', '推荐', '算法',
    '识别', '语音', '画质', '增强', '总结', '摘要', '搜索', '滤镜',
    '特效', '美颜', '剪辑', '生成', '理解', '分析', '优化'
]

# 情感推断函数
def infer_sentiment(rating):
    if rating <= 2:
        return "negative"
    elif rating == 3:
        return "neutral"
    else:
        return "positive"

def is_pain_point(content, rating):
    pain_keywords = ['垃圾', '恶心', '难用', '卡顿', '闪退', 'bug', 
                     '广告太多', '无法', '不能', '失望', '卸载', '恶心',
                     '讨厌', '烦', '糟', '差', '烂']
    if rating <= 2 and any(kw in content for kw in pain_keywords):
        return True
    return False

def is_expectation(content):
    expect_keywords = ['希望', '想要', '如果能', '建议增加', '期待', 
                       '需要', '最好', '为什么不', '怎么没有', '建议']
    return any(kw in content for kw in expect_keywords)

# 扩展App Store评论收集
apps = [
    ("抖音", 1142110895),
    ("B站", 736536082),
    ("快手", 440948110),
    ("腾讯视频", 458318329),
    ("爱奇艺", 393765873),
    ("优酷", 336475695),
    ("芒果TV", 629774185),
    ("YouTube", 544007664),
    ("TikTok", 835599320),
    ("Netflix", 363590051),
    ("Instagram", 389801255),
    ("Snapchat", 447188370)
]

all_vocs = []
target_per_app = 50  # 每个应用目标50条

for app_name, app_id in apps:
    print(f"\n正在收集: {app_name} (ID: {app_id})")

    try:
        # 使用iTunes API获取评论
        url = f"https://itunes.apple.com/rss/customerreviews/page=1/id={app_id}/sortby=mostrecent/json?l=en&&cc=cn"
        response = requests.get(url, timeout=10)

        if response.status_code != 200:
            print(f"  无法获取 {app_name}: HTTP {response.status_code}")
            continue

        data = response.json()
        entries = data.get('feed', {}).get('entry', [])

        if not entries:
            print(f"  {app_name}: 无评论数据")
            continue

        app_vocs = []
        for entry in entries[:target_per_app]:
            # 提取评论内容
            content = entry.get('content', {}).get('label', '')
            if not content:
                continue

            # 提取星级
            rating_label = entry.get('im:rating', {}).get('label', '3')
            try:
                rating = int(rating_label)
            except:
                rating = 3

            # 检查AI相关性
            is_ai_related = any(kw in content for kw in AI_KEYWORDS)

            # 推断情感
            sentiment = infer_sentiment(rating)
            if is_pain_point(content, rating):
                sentiment = "pain_point"
            elif is_expectation(content):
                sentiment = "expectation"

            voc = {
                "id": f"voc_appstore_{len(all_vocs)+len(app_vocs)+1:06d}",
                "platform": "app_store",
                "source_url": f"https://apps.apple.com/app/id{app_id}",
                "author": entry.get('author', {}).get('name', {}).get('label', 'anonymous'),
                "date": entry.get('updated', {}).get('label', 'unknown'),
                "content": content,
                "context": f"{app_name} App Store评论",
                "sentiment": sentiment,
                "category": "other",
                "ai_related": is_ai_related,
                "metadata": {
                    "rating": rating,
                    "app_name": app_name,
                    "app_id": app_id
                }
            }

            app_vocs.append(voc)

        all_vocs.extend(app_vocs)
        print(f"  {app_name}: 收集 {len(app_vocs)} 条")

        # 避免请求过快
        time.sleep(1)

    except Exception as e:
        print(f"  {app_name} 出错: {str(e)}")

# 保存结果
output = {
    "collection_date": datetime.now().isoformat(),
    "total_collected": len(all_vocs),
    "platform": "app_store",
    "voc_entries": all_vocs
}

output_path = "/Users/zhijian/workspace/video_ai_research/raw_voc/app_store/voc_appstore_batch1.json"
with open(output_path, 'w', encoding='utf-8') as f:
    json.dump(output, f, ensure_ascii=False, indent=2)

print(f"\n{'='*50}")
print(f"总收集: {len(all_vocs)} 条VOC")
print(f"已保存: {output_path}")

# 统计情感分布
sentiments = Counter(v['sentiment'] for v in all_vocs)
print(f"\n情感分布:")
for sent, count in sentiments.most_common():
    print(f"  {sent:15s}: {count:4d} ({count/len(all_vocs)*100:.1f}%)")

🚨 问题发现

pain_point 和 expectation 的识别率极低！

  • pain_point：需要同时满足 rating <= 2 且包含痛点关键词，条件太严格
  • expectation：App Store评论中主动提期望的用户很少，大部分是抱怨或好评

情感标记的局限性

问题分析

标记规则的问题

规则 问题 实际效果
rating <= 2 → negative 正确 大部分负面被识别
rating >= 4 → positive 正确 大部分正面被识别
rating == 3 → neutral 基本正确 少量
pain_point识别 条件太严格 0条被识别
expectation识别 用户很少主动提期望 0条被识别

关键缺陷

1. 没有真正的"期望"类VOC

  • App Store评论场景用户主要是投诉或表扬
  • 主动说"我希望有XX功能"的用户很少
  • 这导致期望类AI诉求被严重低估

2. pain_point识别失败

  • 代码逻辑：rating <= 2 AND 包含痛点关键词
  • 实际：很多1-2星评论只是简单骂"垃圾""难用"，没包含预设的关键词
  • 预设关键词列表太短，覆盖面不足

3. 情感推断过于粗糙

  • 仅基于星级，没有分析文本内容
  • 有些4-5星评论也可能包含改进建议（混合情感）
  • 有些1-2星评论可能是误操作或网络问题，并非真实负面

总结：情感分析的真相

核心结论

实际执行的方法

阶段 方法 工具
收集阶段 基于星级推断 + 关键词规则 Python函数
分析阶段 统计情感分布 Counter

核心问题

情感分析是"推断"而非"真正分析"：

Python 
# 实际代码：简单的if-else规则
def infer_sentiment(rating):
    if rating <= 2: return "negative"
    elif rating == 3: return "neutral"
    else: return "positive"

这不是真正的NLP情感分析，而是：

  • 星级映射（1-2星=负面，3星=中性，4-5星=正面）
  • 极少量的关键词规则修正

与真正情感分析的差距

维度 实际做的 应该做的
分析方法 星级映射 BERT/GPT情感分类
粒度 整条评论一个标签 句子级/方面级情感
期望识别 关键词匹配（失败） 意图识别模型
痛点识别 星级+关键词（失败） 细粒度情感+主题抽取
混合情感 无法处理 方面级情感分析

对报告的影响

由于情感分析的粗糙性，报告存在以下偏见：

📉 负面占比过高

67.5%的负面率，因为App Store评论场景本身就是投诉为主

❌ 期望类VOC缺失

只有1条，严重低估了用户的AI功能期望

🔄 没有混合情感

无法识别"整体好评但有个小建议"的评论
