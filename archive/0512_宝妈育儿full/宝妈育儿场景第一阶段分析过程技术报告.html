# 特殊格式处理函数
def parse_count(value):
    if not value: return 0
    value = str(value).strip()
    if '万' in value:
        return int(float(
            value.replace('万+', '').replace('万', '')
        ) * 10000)
    return int(value)

# 应用转换
df['liked_count_num'] = df['liked_count'].apply(parse_count)
df['total_interaction'] = (
    df['liked_count_num'] + df['collected_count_num'] + 
    df['comment_count_num'] + df['share_count_num']
)

# 合并文本字段
df['text'] = df['title'].fillna('') + ' ' + df['desc'].fillna('')

# 匹配算法：只要匹配任意一个关键词即归入该主题
def match_topic(text, topics):
    matched = []
    for topic, keywords in topics.items():
        for kw in keywords:
            if kw in text:
                matched.append(topic)
                break
    return matched

marketing_signals = ['好物', '推荐', '测评', '必买', '清单',

             '我的', '我家']

def classify_content(text):
    has_marketing = any(s in text for s in marketing_signals)
    has_ugc = any(s in text for s in ugc_signals)
    
    if has_marketing and not has_ugc: return '纯营销'
    elif has_ugc and not has_marketing: return '纯UGC'
    elif has_marketing and has_ugc: return '混合'
    else: return '中性'

Step 6: 产品机会映射方法
评估矩阵
维度权重评估标准

优先级定义
P0: 需求强 + 缺口大 + 可行 → 立即做
    例: 睡眠作息(7.5%覆盖, 13,308互动) → 智能睡眠助手
    
P1: 需求中 + 缺口中 → 短期规划
    例: 喂养辅食(13.2%覆盖, 15,611互动) → 辅食添加决策助手
    
P2: 需求弱或技术难 → 中期探索
    例: 职场平衡(8.0%覆盖, 14,540互动) → 背奶提醒+社区
方法论总结
核心原则
1. 不预设需求：用场景+情绪词搜索，而非功能名。当前数据搜索词为身份标签（备孕/孕期/产后），导致内容偏向生活记录，需优化为"夜醒频繁"、"宝宝发烧"等场景词。

2. 多维度交叉验证：主题覆盖率 + 互动表现 + 需求信号 + 内容质量，四维度交叉识别真实需求。

3. 关注异常信号：低覆盖+高互动=供给缺口。睡眠作息7.5%覆盖但13,308互动，说明需求强烈但内容不足。

4. UGC优先：纯UGC内容（39.9%）平均评论901，高于纯营销（468），优先采用UGC内容。

5. 从情绪到功能：痛点词（难/焦虑/累）→ 行为词（查/搜/问）→ 期望词（科学/简单/解决）→ 功能机会。
